# Convert Order Date column into datetime format safely (invalid values become NaT)
df['Order Date'] = pd.to_datetime(df['Order Date'], errors='coerce')

# Remove rows where Order Date is invalid (NaT values)
df = df.dropna(subset=['Order Date'])

# Create Month-Year column for grouping sales on monthly basis
df['Month_Year'] = df['Order Date'].dt.to_period('M')

# Group data by Month-Year and calculate total sales for each month
monthly_sales = df.groupby('Month_Year')['Sales'].sum()

# Display first 5 records of monthly sales
print(monthly_sales.head())
